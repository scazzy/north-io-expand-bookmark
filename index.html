    (function() {
      !function() {
          if (!window.location.hostname.includes('to.thenorth.io')) {
              alert("You're not on https://to.thenorth.io. Click on the bookmark on North's Team Goals page. If you haven't bookmarked it yet, please do so!")
              return;
          }
          var e = document.createElement("style");
          e.textContent = `
            body.present-mode .main-container-fluid {
            max-width: none !important;
         }
         .goals-wrapper .goals-container.col-md-10 {
            flex: 0 0 100% !important;
            max-width: 100% !important;
         }
          `;
          document.head.appendChild(e);
          document.querySelector(".sidebar").classList.toggle("opened");
          document.querySelector(".sidebar").classList.toggle("minimized");
          document.querySelector("body").classList.toggle("present-mode");
      }();
    })();
